api: "POST /createCampaign"
description: "Creates a new campaign with contacts and configuration settings."

## Overview

Creates a new campaign with contacts and configuration settings. Optionally starts the campaign immediately.

---

## Headers

<ParamField header="Cookie" type="string" required>
  Authentication token from login (httpOnly cookie)
  
  Example: `token=your_jwt_token_here`
</ParamField>

  application/json
</ParamField>

  application/json, text/plain, */*
</ParamField>

  Origin of the request (e.g., https://vb-seamless-dev-backend.oriserve.com/)

## Request Body

<ParamField body="campaignName" type="string" required>
  Unique name for the campaign
  
  Example: `Q4 Sales Campaign`
</ParamField>

<ParamField body="contactName" type="string" required>
  Contact ID to fetch contacts from (MongoDB ObjectId)
  
  Example: `60d5ec49c1234567890abcde`
</ParamField>

<ParamField body="campaignType" type="string">
  Campaign execution type
  
  **Values:** `fixed-date` | `continuous` (default: `continuous`)
</ParamField>

<ParamField body="campaignStartDate" type="string">
  Campaign start date in ISO 8601 format
  
  Example: `2024-01-15T00:00:00Z`
</ParamField>

<ParamField body="campaignEndDate" type="string">
  Campaign end date in ISO 8601 format
  
  Example: `2024-01-31T23:59:59Z`
</ParamField>

<ParamField body="campaignStartTimezone" type="string">
  Timezone for start time (default: `Asia/Calcutta`)
  
  Example: `America/New_York`
</ParamField>

<ParamField body="campaignStartHour" type="number" required>
  Hour to start calling (0-23)
  
  Example: `9`
</ParamField>

<ParamField body="campaignStartMinute" type="number" required>
  Minute to start calling (0-59)
  
  Example: `0`
</ParamField>

<ParamField body="campaignEndTimezone" type="string">
  Timezone for end time (default: `Asia/Calcutta`)
  
  Example: `America/New_York`
</ParamField>

<ParamField body="campaignEndHour" type="number" required>
  Hour to stop calling (0-23)
  
  Example: `18`
</ParamField>

<ParamField body="campaignEndMinute" type="number" required>
  Minute to stop calling (0-59)
  
  Example: `0`
</ParamField>

<ParamField body="agentId" type="string" required>
  MongoDB ObjectId of the script/agent
  
  Example: `60d5ec49c1234567890abcdf`
</ParamField>

<ParamField body="isScheduledCall" type="boolean">
  Enable scheduled calls at specific datetime
  
  Default: `false`
</ParamField>

<ParamField body="scheduledCallDateTime" type="string">
  DateTime for scheduled calls (required if `isScheduledCall` is true)
  
  Example: `2024-01-20T14:00:00Z`
</ParamField>

<ParamField body="isContextBasedCall" type="boolean">
  Enable context-based calling using previous call history
  
  Default: `false`
</ParamField>

<ParamField body="recentCallContextCount" type="number">
  Number of recent calls to use for context (required if `isContextBasedCall` is true)
  
  Example: `3`
</ParamField>

<ParamField body="redialConfig" type="object">
  Redial configuration for failed calls
  
  <Expandable title="Redial Config Properties">
    <ParamField body="enabled" type="boolean">
      Enable redial functionality
    </ParamField>
    
    <ParamField body="callBackHandlingOnly" type="boolean">
      Only redial callback requests
    </ParamField>
    
    <ParamField body="campaignWiseLimit" type="number">
      Maximum redials per contact in campaign
    </ParamField>
    
    <ParamField body="dailyLimit" type="number">
      Maximum redials per contact per day
    </ParamField>
    
    <ParamField body="callEntities" type="array">
      Array of call entity filters
    </ParamField>
    
    <ParamField body="terminalConditions" type="array">
      Array of conditions that stop redials
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="PhoneNumber" type="array">
  Array of phone numbers to associate with campaign
  
  Example: `["+1234567890", "+0987654321"]`
</ParamField>

## Notes

- Campaign name must be unique
- Agent/Script must exist before campaign creation
- Contact must be loaded with data before campaign starts
- Timezone defaults to Asia/Calcutta if not specified
